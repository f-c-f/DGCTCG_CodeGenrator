inline void vfrsqrt(float *inport, float *outport)
{
    asm(
        "li t0, 4\n\t"
        "vsetvli zero, t0, e32, m1\n\t"
        "vle.v v0, (%1)\n\t"
        "vfsqrt.v v0, v0\n\t"
        "vfmv.v.f v1, %[one]\n\t"
        "vfdiv.vv v0, v1, v0\n\t"
        "vse.v v0, (%0)\n\t"
        :
        : "r"(outport), "r"(inport), [one] "f"(1.0f)
        : "v0", "t0");
}
