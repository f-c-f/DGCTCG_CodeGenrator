#include <stdio.h>
#include <stdlib.h>
void binaryFileAppend_i8(const char* filename, char* data, int size) {
    FILE* file = fopen(filename, "ab");
    if (file == NULL) {
        fprintf(stderr, "Error opening file %s for appending.\n", filename);
        exit(1);
    }
    size_t itemsWritten = fwrite(data, sizeof(char) * size, 1, file);
    if (itemsWritten != 1) {
        fprintf(stderr, "Error appending to file %s.\n", filename);
        fclose(file);
        exit(1);
    }
    fclose(file);
}
