cmake_minimum_required (VERSION 3.21)

# 房쮢쫉模뒫싐謄뻕맪纜?if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -Wall -Wextra -g -O0")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -g -O0 -std=c++17")
else()
    set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -Wall -Wextra -O2")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -O2 -std=c++17")
endif()

project ("ParserForSimulink")

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ../exe/ParserForSimulink)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY_DEBUG ../exe/ParserForSimulink)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY_RELEASE ../exe/ParserForSimulink)
set_property(GLOBAL PROPERTY USE_FOLDERS ON)
set_property(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VS_STARTUP_PROJECT ParserForSimulink)


file(GLOB SOURCE_Common ../Common/*.cpp ../Common/*.h ../Common/*.hpp)
file(GLOB SOURCE_ILBasic ../ILBasic/*.cpp ../ILBasic/*.h)
file(GLOB SOURCE_MIRBasic ../MIRBasic/*.cpp ../MIRBasic/*.h)
file(GLOB SOURCE_ParserForSimulink ../ParserForSimulink/*.cpp ../ParserForSimulink/*.h)

add_executable (ParserForSimulink 
	${SOURCE_Common}
	${SOURCE_ILBasic}
	${SOURCE_MIRBasic}
	${SOURCE_ParserForSimulink}
)

source_group("Common" FILES ${SOURCE_Common})
source_group("ILBasic" FILES ${SOURCE_ILBasic})
source_group("MIRBasic" FILES ${SOURCE_MIRBasic})
source_group("ParserForSimulink" FILES ${SOURCE_ParserForSimulink})
